import useGlobalData from '@docusaurus/useGlobalData';

export function DrinksStats() {
  const stats = useGlobalData()['drinks-stats']?.default;

  if (!stats) {
    return null;
  }

  return (
    <>
      <div className='stats'>
        <div className='statBox'>
          <div className='label'>
            <p>Number of drinks</p>
          </div>
          <div className='value'>
            <p>{stats.fileCount}</p>
          </div>
        </div>
        <div className='statBox'>
          <div className='label'>
            <p>Average rating</p>
          </div>
          <div className='value'>
            <p>{stats.averageRating.toFixed(1)}</p>
          </div>
        </div>
      </div>

      <br /><br />

      <div>
        {stats.ratings.map(({ stars, percentage }) => (
          <p key={stars}>
            <span className='rating'>{stars}</span>&nbsp;&nbsp;&nbsp;
            <span className='value'>{percentage}%</span>
          </p>
        ))}
      </div>
    </>
  );
}
