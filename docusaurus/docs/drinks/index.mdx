import useGlobalData from '@docusaurus/useGlobalData';
import syrupsData from '@site/static/syrups.json';

export function DrinksStats() {
  const stats = useGlobalData()['drinks-stats']?.default;

  if (!stats) {
    return null;
  }

  const statBoxes = [
    { label: 'Number of drinks', value: stats.fileCount },
    { label: 'Average rating', value: stats.averageRating.toFixed(1) },
    { label: 'Top base spirit', value: stats.topSpirit },
    { label: 'Top glassware', value: stats.topGlassware },
  ];

  const ratingLabels = {
    '★☆☆☆☆': 'Terrible',
    '★★☆☆☆': 'Poor',
    '★★★☆☆': 'Acceptable',
    '★★★★☆': 'Remarkable',
    '★★★★★': 'Outstanding',
  };

  return (
    <>
      <div className='boxes'>
        {statBoxes.map(({ label, value }) => (
          <div className='statBox' key={label}>
            <div className='label'>
              <p>{label}</p>
            </div>
            <div className='value'>
              <p>{value}</p>
            </div>
          </div>
        ))}
      </div>

      <br />
      <p><a href='/drinks.csv' download>Download CSV</a></p>
      <br />

      <div>
        {stats.ratings.map(({ stars, percentage }) => (
          <p key={stars}>
            <span className='rating' title={ratingLabels[stars]}>
              {stars}
            </span>
            <span className='value percentage'>{percentage}%</span>
          </p>
        ))}
      </div>
    </>
  );
}

export function DrinksCards() {
  const cardBoxes = syrupsData;

  return (
    <div className='boxes'>
      {cardBoxes.map(({ title, ingredients }) => (
        <div className='cardBox' key={title}>
          <div className='label'>
            {title}
          </div>
          <div>
            {ingredients.map((line) => (
              <div key={line}>{line}</div>
            ))}
          </div>
        </div>
      ))}
    </div>
  );
}
