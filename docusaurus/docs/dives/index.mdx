import useGlobalData from '@docusaurus/useGlobalData';
import certsData from '@site/static/certs.json';
import { useColorMode } from '@docusaurus/theme-common';

export function DivesStats() {
  const stats = useGlobalData()['dives-stats']?.default;

  if (!stats) {
    return null;
  }

  const statBoxes = [
    { label: 'Number of dives', value: stats.rowCount },
    { label: 'Total dive time', value: stats.totalTime },
    { label: 'Maximum dive time', value: stats.maxTime },
    { label: 'Average dive time', value: stats.averageTime },
    { label: 'Maximum depth', value: stats.maxDepth },
    { label: 'Minimum temperature', value: stats.minTemperature },
    { label: 'Maximum temperature', value: stats.maxTemperature },
    { label: 'Number of dive sites', value: stats.diveSites },
  ];

  return (
    <div className='boxes'>
      {statBoxes.map(({ label, value }) => (
        <div className='statBox' key={label}>
          <div className='label'>
            <p>{label}</p>
          </div>
          <div className='value'>
            <p>{value}</p>
          </div>
        </div>
      ))}
    </div>
  );
}

export function DivesCards() {
  const cardBoxes = certsData;
  const { colorMode } = useColorMode();

  const getLogo = (logo) => {
    if (colorMode === 'light') {
      return logo
        .replace('/img/padi.svg', '/img/padi-black.svg')
        .replace('/img/ssi.svg', '/img/ssi-black.svg');
    }
    return logo;
  };

  return (
    <div className='boxes'>
      {cardBoxes.map(({ name, id, date, logo }) => (
        <div className='cardBox certDetails' key={id}>
          <div className='label'>
            <img src={getLogo(logo)} className='logo'></img>
            {name}
          </div>
          <div className='id'>{id}</div>
          <div>{date}</div>
        </div>
      ))}
    </div>
  );
}
