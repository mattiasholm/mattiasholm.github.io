import certsData from '@site/static/certs.json';
import { useColorMode } from '@docusaurus/theme-common';

export function DivesStats() {
  const statBoxes = [
    { label: 'Number of dives', value: '132' },
    { label: 'Total time', value: '102 hours, 56 minutes' },
    { label: 'Max depth', value: '39,5 meters' },
    { label: 'Max time', value: '71 minutes' },
  ];

  return (
    <div className='boxes'>
      {statBoxes.map(({ label, value }) => (
        <div className='statBox' key={label}>
          <div className='label'>
            <p>{label}</p>
          </div>
          <div className='value'>
            <p>{value}</p>
          </div>
        </div>
      ))}
    </div>
  );
}

export function DivesCards() {
  const cardBoxes = certsData;
  const { colorMode } = useColorMode();

  const getLogo = (logo) => {
    if (colorMode === 'light') {
      return logo
        .replace('/img/padi.svg', '/img/padi-black.svg')
        .replace('/img/ssi.svg', '/img/ssi-black.svg');
    }
    return logo;
  };

  return (
    <div className='boxes'>
      {cardBoxes.map(({ name, id, date, logo }) => (
        <div className='cardBox certDetails' key={id}>
          <div className='label'>
            <img src={getLogo(logo)} className='logo'></img>
            {name}
          </div>
          <div className='id'>{id}</div>
          <div>{date}</div>
        </div>
      ))}
    </div>
  );
}
