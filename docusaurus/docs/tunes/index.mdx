import DocCardList from '@theme/DocCardList';
import useGlobalData from '@docusaurus/useGlobalData';

export function TunesStats() {
  const stats = useGlobalData()['tunes-stats']?.default;

  if (!stats) {
    return null;
  }

  const statBoxes = [
    { label: 'Number of tunes', value: stats.fileCount },
    { label: 'Top tune type', value: stats.topTuneType },
    { label: 'Top key signature', value: stats.topKeySignature },
  ];

  return (
    <div className='boxes'>
      {statBoxes.map(({ label, value }) => (
        <div className='statBox' key={label}>
          <div className='label'>
            <p>{label}</p>
          </div>
          <div className='value'>
            <p>{value}</p>
          </div>
        </div>
      ))}
    </div>
  );
}
